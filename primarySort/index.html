func merge(left int, right, mid int, array []int, temp []int) {
    var (
        i int = left
        j int = mid + 1
        k int = 0
    )
    for ; i <= mid && j <= right; k ++ {
        if array[i] < array[j] {
            temp[k] = array[i]
            i += 1
        } else {
            temp[k] = array[j]
            j += 1
        }
    }
    for ; i <= mid; i, k = i + 1, k + 1 {
        temp[k] = array[i]
    }
    for ; j <= right; j, k = j + 1, k + 1 {
        temp[k] = array[j]
    }
    for l := left; l <= right; l ++ { // å°†æ’åºåçš„æ•°æ®æ”¾å›åŸæ•°ç»„

func sort(left, right int, array []int) int {

        mark int = array[left] // åŸºå‡†çš„æ•°å€¼

func heap(array []int, i, length int) {
    var (
        left int = i * 2 + 1
        right int = i * 2 + 2
        max int = i
    )
    if left < length && array[max] < array[left] {
        max = left
    }
    if right < length && array[max] < array[right] {
        max = right
    }
    if max != i {
        array[i], array[max] = array[max], array[i]
        heap(array, max, length)
    }
}
