Golang 多线程图片爬取

package main

import (
	"strconv"
	"net/http"
	"io/ioutil"
	"github.com/anaskhan96/soup"
	// . "fmt"
	"sync"
	"runtime"
)

var (
	wg sync.WaitGroup
	foo int
	lock sync.Mutex
	pageNum int
)


func pageSpider(pageUrl string, urlChannle chan string, wait chan int) {
	soup.Header("user-agent", "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/73.0.3683.103 Safari/537.36")
	source, _ := soup.Get(pageUrl)
	for _, bar := range soup.HTMLParse(source).Find("ul", "id", "post-list-posts").FindAll("li") {
		urlChannle <- bar.FindAll("a")[1].Attrs()["href"]
	}
	wait <- 1
}

func save(urlChannle chan string) {
	for i := range urlChannle {
		res, _ := http.Get(i)
		defer res.Body.Close()
		data, _ := ioutil.ReadAll(res.Body)
        // 防止数据异常
		lock.Lock()
		_ = ioutil.WriteFile("/home/tsubasa/图片/photo/" + strconv.Itoa(foo) + ".jpg", data, 0755)
		foo ++
		lock.Unlock()
	}
	defer wg.Done()
}

func main() {
    // 设置运行的 CPU 核数
	runtime.GOMAXPROCS(4)
    // 爬取的页面数量
	pageNum = 100
	urlChannle := make(chan string, 20)
	wait := make(chan int)
	// 调度器
	go func() {
		for i := 0; i < pageNum; i ++ {
			<-wait
		}
		close(urlChannle)
	}()
	for i := 1; i < pageNum + 1; i ++ {
		go pageSpider("https://konachan.net/post?page="+ strconv.Itoa(i) +"&tags=", urlChannle, wait)
	}
	for i := 0; i < 4; i ++ {
		wg.Add(1)
		go save(urlChannle)
	}
	wg.Wait()
}
