func searchNode(node *AVLTreeNode) (*AVLTreeNode, *AVLTreeNode) {
	if node.Left == nil {
		return nil, node
	}
	for {
		if node.Left.Left != nil {
			node = node.Left
		} else {
			return node, node.Left
